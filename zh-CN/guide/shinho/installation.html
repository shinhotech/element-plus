安装与快速上手 #

设置 registry #

npm set registry http://10.211.62.41:4873

安装组件库 #

通过 npm/yarn/pnpm 安装

npm install @shinho-yt/sh-design-plus or
yarn add @shinho-yt/sh-design-plus or
pnpm install @shinho-yt/sh-design-plus

在项目入口文件（一般是 main.js 或者 main.ts）注册组件库 #

import ShDesignPlus from '@shinho-yt/sh-design-plus';
import '@shinho-yt/sh-design-plus/packages/theme/style/index.scss';

createApp(App)
  .use(store)
  .use(router)
  .use(ShDesignPlus)
  .mount('#app');

基于Vite引入 svg 组件库 #

vite.config.js

import { defineConfig } from 'vite';
import vue from '@vitejs/plugin-vue';

// eslint-disable-next-line @typescript-eslint/no-var-requires
const { resolve } = require('path');
const {
  svgBuilder
  // eslint-disable-next-line @typescript-eslint/no-var-requires
} = require('@shinho-yt/sh-design-plus/packages/icons/svgBuilder');

// https://vitejs.dev/config/
export default defineConfig({
  plugins: [vue(), svgBuilder(resolve(__dirname, 'node_modules/@shinho-yt/sh-design-plus/packages/icons/src/svg/'))]
});
基于webpack引入 svg 组件库 #

vue.config.js

module.exports = {
  chainWebpack: config => {
    const svgRule = config.module.rule('svg');

    // 清除已有的所有 loader。
    // 如果你不这样做，接下来的 loader 会附加在该规则现有的 loader 之后。
    svgRule.uses.clear();

    // 添加要替换的 loader
    svgRule
      .use('svg-sprite-loader')
      .loader('svg-sprite-loader')
      .options({
        symbolId: '[name]'
      });
  }
};

main.ts

const requireAll = (requireContext: any) => requireContext.keys().map(requireContext);

const req = require.context('@shinho-yt/sh-design-plus/packages/icons/src/svg/', false, /\.svg$/);

requireAll(req);
